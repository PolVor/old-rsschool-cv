            if (n < 0) {
              return 0 
            }
            
            var result = 0;
            let tr;
            let i = n;
            
            while ( i >= 0) {
            tr = (i/2) * (i + 1) 
            i = i - 1;
            result = result + tr; 
            }
            
            return result; 
          };
